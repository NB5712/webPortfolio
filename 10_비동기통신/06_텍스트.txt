  <script>
    $(function () {
      $.ajax({
        url: "./json/icon.json", // 불러올 JSON 파일 경로
        dataType: "json",       // JSON 형식으로 응답을 처리함
        success: function (data) { // 성공 시 처리
          console.log(data);
          
          if (data.length > 0) { // data는 배열이고, 각 항목은 하나의 상품 정보
            for (var i in data) {
              //review random count
              var count = Math.floor(Math.random() * 100); // 0 ~ 99 임의 숫자
              var isNew = data[i].isNew ? `<span class="new-badge">NEW</span>` : ""; // new 아이콘

              $(".box")
                .eq(i)
                .append(
                  `
                    <a href="#">
                      <figure>
                        <img src="img/hit/${data[i].url}">
                        <figcaption>
                          <div>${isNew}</div>
                          <h3>${data[i].title}</h3>
                          <p>${data[i].subtitle}</p>
                          <span class="price">${data[i].price}</span>
                          <span class="btn">97 REVIEWS</span>
                        </figcaption>
                      </figure>
                    </a>
                  ` );
            }
          }
        },
        error: function (err) { // 실패 시 처리
          console.error('로컬 JSON 불러오기 실패', err);
        }
      });
    });
  </script>
