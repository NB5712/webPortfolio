$(function(){
  // 자주 수정이 발생하는 요소들 전역변수로 설정
  const frame = "section";
  const box = "article";
  const speed = "0.5s";
  let grid;

  // 초기화 함수 호출
  init();

  // 초기화하는 함수 정수
  function init(){
    grid = new Isotope(frame, { // 배치할 요소를 감싸고 있는 부모 요소명
      itemSelector: box, // 배치할 요소명
      columnWidth: box, // 너빗값을 구할 요소명
      transitionDuration: speed // 화면 재배치시 요소가 움직이는 속도
    });
  }

  // 버튼 이벤트
  $('main').on('click','li',function(e){
    e.preventDefault(); // a 기능 막음
    const sort = $(this).children('a').attr('href');

    // 전체 li 디자인
    $('main ul li').removeClass('on');
    $(this).addClass('on');

    grid.arrange({
      filter: sort
    });
  });
});
